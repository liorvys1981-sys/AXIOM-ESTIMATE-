---
apiVersion: v1
kind: Namespace
metadata:
  name: axiom
  labels:
    app.kubernetes.io/name: axiom
    environment: production
    pod-security.kubernetes.io/enforce: restricted
    pod-security.kubernetes.io/warn: restricted

---
# Service account for the API pods
apiVersion: v1
kind: ServiceAccount
metadata:
  name: axiom-api
  namespace: axiom
  labels:
    app: axiom-api
automountServiceAccountToken: false

---
# Service account for Celery workers
apiVersion: v1
kind: ServiceAccount
metadata:
  name: axiom-worker
  namespace: axiom
  labels:
    app: axiom-worker
automountServiceAccountToken: false

---
# Service account for the watchdog
apiVersion: v1
kind: ServiceAccount
metadata:
  name: axiom-watchdog
  namespace: axiom
  labels:
    app: axiom-watchdog
automountServiceAccountToken: false

---
# Watchdog needs read access to pods/deployments to detect stale workers
apiVersion: rbac.authorization.k8s.io/v1
kind: Role
metadata:
  name: axiom-watchdog-role
  namespace: axiom
rules:
  - apiGroups: ["apps"]
    resources: ["deployments", "replicasets"]
    verbs: ["get", "list", "watch"]
  - apiGroups: [""]
    resources: ["pods"]
    verbs: ["get", "list", "watch", "delete"]

---
apiVersion: rbac.authorization.k8s.io/v1
kind: RoleBinding
metadata:
  name: axiom-watchdog-binding
  namespace: axiom
subjects:
  - kind: ServiceAccount
    name: axiom-watchdog
    namespace: axiom
roleRef:
  kind: Role
  name: axiom-watchdog-role
  apiGroup: rbac.authorization.k8s.io
