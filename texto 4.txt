version: '3.8'

# ============================================================
# AXIOM ESTIMATE — Full Stack Orchestration
# Services: PostgreSQL, Redis, FastAPI, Celery Worker, Watchdog
# ============================================================

x-common-env: &common-env
  DATABASE_URL:       postgresql+asyncpg://axiom:${POSTGRES_PASSWORD:-changeme}@db:5432/axiom_db
  SYNC_DATABASE_URL:  postgresql://axiom:${POSTGRES_PASSWORD:-changeme}@db:5432/axiom_db
  REDIS_URL:          redis://:${REDIS_PASSWORD:-}@redis:6379/0
  CELERY_APP:         axiom.agent_core
  CELERY_CONCURRENCY: "4"
  STRIPE_SECRET_KEY:  ${STRIPE_SECRET_KEY}
  STRIPE_WEBHOOK_SECRET: ${STRIPE_WEBHOOK_SECRET}
  OPENAI_API_KEY:     ${OPENAI_API_KEY}
  SECRET_KEY:         ${SECRET_KEY:-insecure-dev-key-replace-in-prod}
  ENVIRONMENT:        ${ENVIRONMENT:-production}
  LOG_LEVEL:          ${LOG_LEVEL:-info}

services:

  # ----------------------------------------------------------
  # PostgreSQL 15 — Multi-tenant relational store
  # ----------------------------------------------------------
  db:
    image: postgres:15-alpine
    restart: unless-stopped
    environment:
      POSTGRES_DB:       axiom_db
      POSTGRES_USER:     axiom
      POSTGRES_PASSWORD: ${POSTGRES_PASSWORD:-changeme}
    volumes:
      - postgres_data:/var/lib/postgresql/data
      - ./db/schema.sql:/docker-entrypoint-initdb.d/01_schema.sql:ro
    ports:
      - "127.0.0.1:5432:5432"   # Bind to loopback only — not exposed to internet
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U axiom -d axiom_db"]
      interval: 10s
      timeout: 5s
      retries: 5
      start_period: 10s
    deploy:
      resources:
        limits:
          memory: 1G

  # ----------------------------------------------------------
  # Redis 7 — Celery broker + result backend
  # ----------------------------------------------------------
  redis:
    image: redis:7-alpine
    restart: unless-stopped
    command: >
      redis-server
      --maxmemory 512mb
      --maxmemory-policy allkeys-lru
      --save 60 1000
      --appendonly yes
    volumes:
      - redis_data:/data
    ports:
      - "127.0.0.1:6379:6379"
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 3s
      retries: 5
    deploy:
      resources:
        limits:
          memory: 600M

  # ----------------------------------------------------------
  # FastAPI — HTTP API node
  # ----------------------------------------------------------
  api:
    build:
      context: .
      dockerfile: docker/Dockerfile.gpu
    restart: unless-stopped
    command: >
      uvicorn api.main:app
      --host 0.0.0.0
      --port 8000
      --workers 2
      --loop uvloop
      --http httptools
      --log-level ${LOG_LEVEL:-info}
    ports:
      - "8000:8000"
    environment:
      <<: *common-env
    depends_on:
      db:
        condition: service_healthy
      redis:
        condition: service_healthy
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8000/health"]
      interval: 20s
      timeout: 5s
      retries: 3
      start_period: 20s
    deploy:
      resources:
        limits:
          memory: 1G

  # ----------------------------------------------------------
  # Celery Worker — GPU-bound async task processor
  # ----------------------------------------------------------
  celery_worker:
    build:
      context: .
      dockerfile: docker/Dockerfile.gpu
    restart: unless-stopped
    command: >
      celery -A axiom.agent_core worker
      --loglevel=${LOG_LEVEL:-info}
      --concurrency=4
      --queues=celery,estimate,claims,total_loss,lien,audit,cpo,gpu_resell
      --without-gossip
      --without-mingle
      --max-tasks-per-child=100
    environment:
      <<: *common-env
      NVIDIA_VISIBLE_DEVICES: all
      NVIDIA_DRIVER_CAPABILITIES: compute,utility,video
    depends_on:
      db:
        condition: service_healthy
      redis:
        condition: service_healthy
    deploy:
      resources:
        reservations:
          devices:
            - driver: nvidia
              count: all
              capabilities: [gpu, compute, utility]
        limits:
          memory: 8G

  # ----------------------------------------------------------
  # Celery Beat — Periodic task scheduler
  # ----------------------------------------------------------
  celery_beat:
    build:
      context: .
      dockerfile: docker/Dockerfile.gpu
    restart: unless-stopped
    command: >
      celery -A axiom.agent_core beat
      --loglevel=${LOG_LEVEL:-info}
      --scheduler django_celery_beat.schedulers:DatabaseScheduler
    environment:
      <<: *common-env
    depends_on:
      db:
        condition: service_healthy
      redis:
        condition: service_healthy
    deploy:
      resources:
        limits:
          memory: 256M

  # ----------------------------------------------------------
  # Watchdog — Self-healing monitor (no GPU needed)
  # ----------------------------------------------------------
  watchdog:
    build:
      context: .
      dockerfile: docker/Dockerfile.gpu
    restart: unless-stopped
    command: python3 -m axiom.self_healing
    environment:
      <<: *common-env
      WATCHDOG_POLL_SECS:       "15"
      MEMORY_WARN_PCT:          "75"
      MEMORY_CRITICAL_PCT:      "90"
      QUEUE_DEPTH_WARN:         "50"
      QUEUE_DEPTH_CRITICAL:     "200"
      MAX_RESTART_ATTEMPTS:     "5"
      STALE_WORKER_THRESHOLD_SECS: "300"
    depends_on:
      redis:
        condition: service_healthy
    deploy:
      resources:
        limits:
          memory: 256M

  # ----------------------------------------------------------
  # Flower — Celery monitoring UI (dev/staging only)
  # ----------------------------------------------------------
  flower:
    build:
      context: .
      dockerfile: docker/Dockerfile.gpu
    restart: unless-stopped
    command: >
      celery -A axiom.agent_core flower
      --port=5555
      --basic_auth=${FLOWER_USER:-admin}:${FLOWER_PASSWORD:-changeme}
    ports:
      - "127.0.0.1:5555:5555"
    environment:
      <<: *common-env
    depends_on:
      - redis
    profiles:
      - dev    # Only started when: docker compose --profile dev up
    deploy:
      resources:
        limits:
          memory: 256M

# ============================================================
# Named volumes — persisted across container restarts
# ============================================================
volumes:
  postgres_data:
    driver: local
  redis_data:
    driver: local

# ============================================================
# Network isolation
# ============================================================
networks:
  default:
    driver: bridge
    ipam:
      config:
        - subnet: 172.28.0.0/16
